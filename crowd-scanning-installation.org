* Context

This document provides a step-by-step installation guide for the *FACES 2 Crowd Scanning* application on NVIDIA Jetson Orin NX devices.

#+begin_quote
The recipe uses absolute paths and can be executed from any directory on the Jetson device.
All code blocks are designed to be idempotent where possible.
#+end_quote

* Phase 2: Project Setup

** Step 2.1: Clone repository

#+begin_src bash :results output replace :exports both :session f2cs
  set -Eeuo pipefail

  TARBALL_URL="https://github.com/${REPO_OWNER}/${REPO_NAME}/archive/refs/tags/${RELEASE_TAG}.tar.gz"
  TARBALL_PATH="${DOWNLOAD_DIR}/${REPO_NAME}-${RELEASE_TAG}.tar.gz"
  EXTRACT_DIR="${DOWNLOAD_DIR}/${REPO_NAME}-${RELEASE_TAG}-extract-${STAMP}"

  mkdir -p "$DOWNLOAD_DIR"

  # --- Download ---
  if [ -f "$TARBALL_PATH" ]; then
      echo "Tarball already downloaded: $TARBALL_PATH"
  else
      echo "Downloading: $TARBALL_URL"
      if [ -n "${GITHUB_TOKEN:-}" ]; then
          curl -L -H "Authorization: Bearer $GITHUB_TOKEN" -o "$TARBALL_PATH" "$TARBALL_URL"
      else
          curl -L -o "$TARBALL_PATH" "$TARBALL_URL"
      fi
      echo "Saved to: $TARBALL_PATH"
  fi

  # --- Extract ---
  mkdir -p "$EXTRACT_DIR"
  tar -xzf "$TARBALL_PATH" -C "$EXTRACT_DIR"

  SRC_DIR="$(find "$EXTRACT_DIR" -mindepth 1 -maxdepth 1 -type d | head -n 1 || true)"
  if [ -z "$SRC_DIR" ]; then
      echo "ERROR: could not find extracted source folder in $EXTRACT_DIR"
      ls -la "$EXTRACT_DIR"
      exit 1
  fi
  echo "Extracted source dir: $SRC_DIR"

  # --- Deploy ---
  if [ -d "$PROJECT_DIR" ] && [ -n "$(ls -A "$PROJECT_DIR" 2>/dev/null)" ]; then
      echo "Existing installation detected. Updating (preserving faceme, faceme_env, config, logs)..."
      rsync -a \
          --exclude='faceme' \
          --exclude='faceme_env' \
          --exclude='config' \
          --exclude='logs' \
          "$SRC_DIR/" "$PROJECT_DIR/"
  else
      echo "Fresh installation. Deploying all files..."
      mkdir -p "$PROJECT_DIR"
      rsync -a "$SRC_DIR/" "$PROJECT_DIR/"
  fi

  echo "Deployed to: $PROJECT_DIR"
#+end_src

* Phase 8: Production Deployment (systemd)

** Step 8.1: Create systemd service file

#+begin_quote
*NOTE:* The license key is automatically embedded from =$LICENSE_KEY= (set in Step 5.3).
Systemd services do not inherit environment variables from =.bashrc=.
#+end_quote

#+begin_src bash :results output replace :exports both :session f2cs
  set -euo pipefail

  cat > "$PROJECT_DIR/f2_crowd_scanner.service" <<SERVICEEOF
[Unit]
Description=F2 Crowd Scanning FaceMe Service
After=network.target
StartLimitBurst=3
StartLimitIntervalSec=60

[Service]
Type=simple
WorkingDirectory=/home/nvidia/projects/F2_Crowd_Scanning
ExecStart=/usr/bin/script -q -c "/home/nvidia/projects/F2_Crowd_Scanning/faceme_env/bin/python /home/nvidia/projects/F2_Crowd_Scanning/main.py" /dev/null
Restart=always
RestartSec=10
Environment=FACEME_LICENSE_KEY=$LICENSE_KEY
StandardOutput=journal
StandardError=journal

[Install]
WantedBy=multi-user.target
SERVICEEOF

  echo "Service file created at: $PROJECT_DIR/f2_crowd_scanner.service"
#+end_src

#+begin_src bash :results output replace :exports both :dir /sudo::
  set -euo pipefail

  cp /home/nvidia/projects/F2_Crowd_Scanning/f2_crowd_scanner.service \
     /etc/systemd/system/f2_crowd_scanner.service

  rm /home/nvidia/projects/F2_Crowd_Scanning/f2_crowd_scanner.service

  echo "Service file installed and temp file removed"
#+end_src

#+begin_src bash :results output replace :exports both :dir /sudo::

#+begin_src bash :results output replace :exports both :dir /sudo::
